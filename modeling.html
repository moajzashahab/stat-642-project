data preparation

# convert categorical columns to factors
df$works <- as.factor(df$works)
df$engine <- as.factor(df$engine)
df$Year <- as.factor(df$Year)

# standardize performance within each race and flag extreme outliers
df <- df %>%
  group_by(Year, EventName) %>%   # group by race session (each qualifying event)
  mutate(FastestQ1 = min(Q1_time, na.rm = TRUE),              # fastest lap time in that session
         GapToFastest = Q1_time - FastestQ1,                  # absolute time gap to fastest driver
         PerToFastest = GapToFastest / FastestQ1,             # relative gap (percentage difference)
         is_beyond_107 = ifelse(PerToFastest > 0.07, 1, 0)    # flag laps slower than 107% rule
         ) %>% ungroup()  

# split data out by outliers
df_fast <- df %>% filter(is_beyond_107 == 0)         # within 107%      
df_slow <- df %>% filter(is_beyond_107 == 1)         # beyond 107% 

# center throttle variable
df$Throttle_Centered <- df$FullThrottlePct - mean(df$FullThrottlePct)
